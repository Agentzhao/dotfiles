//Verbs in Vim
d => Delete
c => Change
> => Indent
v => Visual
y => Yank

>3j => Indent current + 3 lines
3j => Move 3 lines down

//Motions in Vim
w => word (move forward by a word)
b => back (back by a word)
2j => down 2 lines

//nouns in vim
iw => "inner word"
it => "inner tag"
i" => "inner quote"
ip => "inner paragraph"
as => "a sentence"
